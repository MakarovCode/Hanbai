def change_status 
        unless ["0", "1", "2", 0, 1, 2].include? params[:status] 
                render status: 411, json: {message: "Estado no permitido"}
                return
        end
        activity = current_user.activities.find params[:id]
        activity.update_attribute :status, params[:status]

        render status: 200, json: {message: "Actividad pasada de estado correctamente"}
end

def index 
        statuses = params[:statuses]
        
        unless statuses.nil?
                unless statuses.kind_of?(Array)
                        statuses = [statuses]
                end
                if statuses.include? 0 || statuses.include? "0"
                        statuses.push nil
                end
        end
        
        @activities = current_user.activities.by_status(statuses)
end
